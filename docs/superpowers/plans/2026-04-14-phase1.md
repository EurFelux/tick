# tick Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the tick MVP — a CLI issue tracker with semantic state-transition commands, SQLite storage, and JSON-first output.

**Architecture:** Single-binary Rust CLI using clap derive for command parsing, rusqlite for SQLite storage (located at `<git-common-dir>/tick/tick.db`), and serde for JSON output. All state transitions go through dedicated subcommands (start/done/close/reopen) with atomic CAS validation. Validators are isolated in a dedicated module.

**Tech Stack:** Rust (stable latest), clap, serde/serde_json, rusqlite (bundled), anyhow/thiserror, chrono. Testing: assert_cmd, tempfile, assert_json_diff, predicates.

use thiserror::Error;

#[derive(Debug, Error)]
pub enum TickError {
    #[error("{0}")]
    NotFound(String),

    #[error("{0}")]
    InvalidArgument(String),

    #[error("{0}")]
    NotInitialized(String),

    #[error("{0}")]
    Conflict(String),

    #[error(transparent)]
    Db(#[from] rusqlite::Error),

    #[error(transparent)]
    Internal(#[from] anyhow::Error),
}

impl TickError {
    pub fn exit_code(&self) -> i32 {
        match self {
            TickError::Internal(_) => 1,
            TickError::NotFound(_) => 2,
            TickError::InvalidArgument(_) => 3,
            TickError::Db(_) => 4,
            TickError::NotInitialized(_) => 5,
            TickError::Conflict(_) => 6,
        }
    }

    pub fn error_code(&self) -> &str {
        match self {
            TickError::Internal(_) => "INTERNAL_ERROR",
            TickError::NotFound(_) => "NOT_FOUND",
            TickError::InvalidArgument(_) => "INVALID_ARGUMENT",
            TickError::Db(_) => "DB_ERROR",
            TickError::NotInitialized(_) => "NOT_INITIALIZED",
            TickError::Conflict(_) => "CONFLICT",
        }
    }
}

pub type Result<T> = std::result::Result<T, TickError>;
```

use crate::error::{Result, TickError};
use rusqlite::Connection;

const MIGRATIONS: &[(&str, &str)] = &[
    ("001_init", include_str!("../../migrations/001_init.sql")),
];

pub fn run_migrations(conn: &Connection) -> Result<()> {
    // Create schema_version table if not exists (bootstrap)
    conn.execute_batch(
        "CREATE TABLE IF NOT EXISTS schema_version (
            version INTEGER PRIMARY KEY,
            applied_at TEXT NOT NULL DEFAULT (datetime('now', 'utc'))
        );"
    )?;

    let current_version: i64 = conn
        .query_row(
            "SELECT COALESCE(MAX(version), 0) FROM schema_version",
            [],
            |row| row.get(0),
        )?;

    for (i, (_name, sql)) in MIGRATIONS.iter().enumerate() {
        let migration_version = (i + 1) as i64;
        if migration_version > current_version {
            conn.execute_batch(sql)?;
            conn.execute(
                "INSERT INTO schema_version (version) VALUES (?1)",
                [migration_version],
            )?;
        }
    }

    Ok(())
}

pub fn schema_version(conn: &Connection) -> Result<i64> {
    let version: i64 = conn
        .query_row(
            "SELECT COALESCE(MAX(version), 0) FROM schema_version",
            [],
            |row| row.get(0),
        )?;
    Ok(version)
}

pub fn expected_version() -> i64 {
    MIGRATIONS.len() as i64
}
```

pub mod migrate;

use crate::error::Result;
use rusqlite::Connection;
use std::path::Path;

pub struct Database {
    conn: Connection,
}

impl Database {
    pub fn open(path: &str) -> Result<Self> {
        let conn = Connection::open(path)?;
        conn.execute_batch(
            "PRAGMA journal_mode=WAL;
             PRAGMA busy_timeout=5000;
             PRAGMA foreign_keys=ON;"
        )?;
        Ok(Database { conn })
    }

    pub fn migrate(&self) -> Result<()> {
        migrate::run_migrations(&self.conn)
    }

    pub fn schema_version(&self) -> Result<i64> {
        migrate::schema_version(&self.conn)
    }

    pub fn conn(&self) -> &Connection {
        &self.conn
    }
}

pub fn list(conn: &Connection, filter: &ListFilter) -> Result<Vec<Issue>> {
    let mut sql = String::from(
        "SELECT id, parent_id, title, description, type, status, priority, resolution, branch, version, created_at, updated_at FROM issues WHERE 1=1"
    );
    let mut param_values: Vec<Box<dyn rusqlite::types::ToSql>> = Vec::new();

    if let Some(ref status) = filter.status {
        sql.push_str(&format!(" AND status = ?{}", param_values.len() + 1));
        param_values.push(Box::new(status.to_string()));
    }
    if let Some(ref issue_type) = filter.issue_type {
        sql.push_str(&format!(" AND type = ?{}", param_values.len() + 1));
        param_values.push(Box::new(issue_type.to_string()));
    }
    if let Some(ref priority) = filter.priority {
        sql.push_str(&format!(" AND priority = ?{}", param_values.len() + 1));
        param_values.push(Box::new(priority.to_string()));
    }
    if let Some(parent_id) = filter.parent_id {
        sql.push_str(&format!(" AND parent_id = ?{}", param_values.len() + 1));
        param_values.push(Box::new(parent_id));
    }
    if filter.root_only {
        sql.push_str(" AND parent_id IS NULL");
    }

    sql.push_str(" ORDER BY id ASC");
    sql.push_str(&format!(" LIMIT ?{}", param_values.len() + 1));
    let limit = if filter.limit == 0 { 50 } else { filter.limit };
    param_values.push(Box::new(limit));
    sql.push_str(&format!(" OFFSET ?{}", param_values.len() + 1));
    param_values.push(Box::new(filter.offset));

    let params_refs: Vec<&dyn rusqlite::types::ToSql> = param_values.iter().map(|b| b.as_ref()).collect();
    let mut stmt = conn.prepare(&sql)?;
    let issues = stmt
        .query_map(params_refs.as_slice(), |row| Ok(row_to_issue(row)))?
        .collect::<std::result::Result<Vec<_>, _>>()?;
    Ok(issues)
}

pub fn update_fields(
    conn: &Connection,
    id: i64,
    title: Option<&str>,
    description: Option<&str>,
    issue_type: Option<&IssueType>,
    priority: Option<&Priority>,
    parent_id: Option<Option<i64>>,
) -> Result<Issue> {
    let mut sets = Vec::new();
    let mut param_values: Vec<Box<dyn rusqlite::types::ToSql>> = Vec::new();

    if let Some(t) = title {
        param_values.push(Box::new(t.to_string()));
        sets.push(format!("title = ?{}", param_values.len()));
    }
    if let Some(d) = description {
        param_values.push(Box::new(d.to_string()));
        sets.push(format!("description = ?{}", param_values.len()));
    }
    if let Some(t) = issue_type {
        param_values.push(Box::new(t.to_string()));
        sets.push(format!("type = ?{}", param_values.len()));
    }
    if let Some(p) = priority {
        param_values.push(Box::new(p.to_string()));
        sets.push(format!("priority = ?{}", param_values.len()));
    }
    if let Some(pid) = parent_id {
        param_values.push(Box::new(pid));
        sets.push(format!("parent_id = ?{}", param_values.len()));
    }

    if sets.is_empty() {
        return get_by_id(conn, id);
    }

    // Always update version and updated_at
    sets.push("version = version + 1".to_string());
    sets.push("updated_at = datetime('now', 'utc')".to_string());

    param_values.push(Box::new(id));
    let sql = format!(
        "UPDATE issues SET {} WHERE id = ?{}",
        sets.join(", "),
        param_values.len()
    );

    let params_refs: Vec<&dyn rusqlite::types::ToSql> = param_values.iter().map(|b| b.as_ref()).collect();
    let rows = conn.execute(&sql, params_refs.as_slice())?;
    if rows == 0 {
        return Err(TickError::NotFound(format!("issue #{id} not found")));
    }

    get_by_id(conn, id)
}

pub fn update_status_atomic(
    conn: &Connection,
    id: i64,
    expected_status: &str,
    new_status: &str,
    resolution: Option<&str>,
    branch: Option<&str>,
    clear_branch: bool,
    clear_resolution: bool,
) -> Result<Issue> {
    let mut sets = vec![
        format!("status = ?1"),
        "version = version + 1".to_string(),
        "updated_at = datetime('now', 'utc')".to_string(),
    ];

    let mut param_values: Vec<Box<dyn rusqlite::types::ToSql>> = vec![
        Box::new(new_status.to_string()),
    ];

    if let Some(r) = resolution {
        param_values.push(Box::new(r.to_string()));
        sets.push(format!("resolution = ?{}", param_values.len()));
    } else if clear_resolution {
        sets.push("resolution = NULL".to_string());
    }

    if let Some(b) = branch {
        param_values.push(Box::new(b.to_string()));
        sets.push(format!("branch = ?{}", param_values.len()));
    } else if clear_branch {
        sets.push("branch = NULL".to_string());
    }

    param_values.push(Box::new(id));
    let id_pos = param_values.len();
    param_values.push(Box::new(expected_status.to_string()));
    let status_pos = param_values.len();

    let sql = format!(
        "UPDATE issues SET {} WHERE id = ?{} AND status = ?{}",
        sets.join(", "),
        id_pos,
        status_pos,
    );

    let params_refs: Vec<&dyn rusqlite::types::ToSql> = param_values.iter().map(|b| b.as_ref()).collect();
    let rows = conn.execute(&sql, params_refs.as_slice())?;
    if rows == 0 {
        // Check if issue exists to distinguish NOT_FOUND from CONFLICT
        match get_by_id(conn, id) {
            Ok(issue) => Err(TickError::Conflict(format!(
                "issue #{id} is currently '{}', expected '{expected_status}'",
                issue.status
            ))),
            Err(e) => Err(e),
        }
    } else {
        get_by_id(conn, id)
    }
}

pub fn count_by_status(conn: &Connection) -> Result<HashMap<String, i64>> {
    let mut map = HashMap::new();
    map.insert("open".to_string(), 0);
    map.insert("in-progress".to_string(), 0);
    map.insert("done".to_string(), 0);
    map.insert("closed".to_string(), 0);

    let mut stmt = conn.prepare("SELECT status, COUNT(*) FROM issues GROUP BY status")?;
    let rows = stmt.query_map([], |row| {
        Ok((row.get::<_, String>(0)?, row.get::<_, i64>(1)?))
    })?;

    for row in rows {
        let (status, count) = row?;
        map.insert(status, count);
    }

    Ok(map)
}

/// Validate that `close` resolution is allowed given current status.
pub fn validate_close_resolution(status: &IssueStatus, resolution: &Resolution) -> Result<()> {
    match (status, resolution) {
        (IssueStatus::Open | IssueStatus::InProgress, Resolution::Resolved) => {
            Err(TickError::InvalidArgument(
                "cannot close as 'resolved' without review; only 'wontfix' is allowed from this status".to_string()
            ))
        }
        _ => Ok(()),
    }
}

/// Check that parent_id does not create a cycle.
pub fn validate_parent_no_cycle(conn: &Connection, issue_id: i64, new_parent_id: i64) -> Result<()> {
    if issue_id == new_parent_id {
        return Err(TickError::InvalidArgument("issue cannot be its own parent".to_string()));
    }

    let mut current = new_parent_id;
    loop {
        let parent: Option<i64> = conn
            .query_row(
                "SELECT parent_id FROM issues WHERE id = ?1",
                [current],
                |row| row.get(0),
            )
            .map_err(|_| TickError::NotFound(format!("issue #{current} not found")))?;

        match parent {
            None => return Ok(()),
            Some(pid) if pid == issue_id => {
                return Err(TickError::InvalidArgument(format!(
                    "setting parent to #{new_parent_id} would create a cycle"
                )))
            }
            Some(pid) => current = pid,
        }
    }
}

/// Check that all depends-on issues are closed(resolved).
fn check_dependencies_resolved(conn: &Connection, issue_id: i64) -> Result<()> {
    let mut stmt = conn.prepare(
        "SELECT il.to_issue_id, i.status, i.resolution
         FROM issue_links il
         JOIN issues i ON i.id = il.to_issue_id
         WHERE il.from_issue_id = ?1 AND il.relation = 'depends-on'"
    )?;

    let deps = stmt.query_map([issue_id], |row| {
        Ok((
            row.get::<_, i64>(0)?,
            row.get::<_, String>(1)?,
            row.get::<_, Option<String>>(2)?,
        ))
    })?;

    for dep in deps {
        let (dep_id, status, resolution) = dep?;
        let is_resolved = status == "closed" && resolution.as_deref() == Some("resolved");
        if !is_resolved {
            return Err(TickError::InvalidArgument(format!(
                "dependency #{dep_id} is not resolved (status: {status})"
            )));
        }
    }

    Ok(())
}

use crate::db::Database;
use crate::error::{Result, TickError};
use std::fs;
use std::process::Command;

pub fn run(db_path: Option<&str>) -> Result<Database> {
    let path = match db_path {
        Some(p) => p.to_string(),
        None => resolve_db_path()?,
    };

    // Create parent directory
    if let Some(parent) = std::path::Path::new(&path).parent() {
        fs::create_dir_all(parent).map_err(|e| {
            TickError::Internal(anyhow::anyhow!("failed to create directory: {e}"))
        })?;
    }

    let db = Database::open(&path)?;
    db.migrate()?;
    Ok(db)
}

pub fn resolve_db_path() -> Result<String> {
    let output = Command::new("git")
        .args(["rev-parse", "--git-common-dir"])
        .output()
        .map_err(|e| TickError::NotInitialized(format!("git not found: {e}")))?;

    if !output.status.success() {
        return Err(TickError::NotInitialized(
            "not a git repository — tick requires git".to_string(),
        ));
    }

    let git_dir = String::from_utf8_lossy(&output.stdout).trim().to_string();
    Ok(format!("{git_dir}/tick/tick.db"))
}

pub fn open_db(db_path: Option<&str>) -> Result<Database> {
    let path = match db_path {
        Some(p) => p.to_string(),
        None => resolve_db_path()?,
    };

    if !std::path::Path::new(&path).exists() {
        return Err(TickError::NotInitialized(
            "tick not initialized — run 'tick init' first".to_string(),
        ));
    }

    let db = Database::open(&path)?;

    // Check schema version
    let current = db.schema_version()?;
    let expected = crate::db::migrate::expected_version();
    if current != expected {
        return Err(TickError::NotInitialized(format!(
            "schema version mismatch (have {current}, need {expected}) — run 'tick init' to upgrade"
        )));
    }

    Ok(db)
}

use crate::db::Database;
use crate::error::{Result, TickError};
use crate::models::*;
use crate::validators;

pub fn create(
    db: &Database, title: &str, description: &str, issue_type: &str,
    priority: &str, parent_id: Option<i64>,
) -> Result<Issue> {
    let it: IssueType = issue_type.parse().map_err(|e: String| TickError::InvalidArgument(e))?;
    let p: Priority = priority.parse().map_err(|e: String| TickError::InvalidArgument(e))?;

    if let Some(pid) = parent_id {
        // Verify parent exists
        db.get_issue(pid)?;
    }

    let id = db.create_issue(title, description, &it, &p, parent_id)?;
    db.get_issue(id)
}

pub fn list(
    db: &Database, status: Option<&str>, issue_type: Option<&str>,
    priority: Option<&str>, parent_id: Option<i64>, root: bool,
    limit: i64, offset: i64,
) -> Result<Vec<Issue>> {
    let filter = crate::db::issues::ListFilter {
        status: status.map(|s| s.parse().map_err(|e: String| TickError::InvalidArgument(e))).transpose()?,
        issue_type: issue_type.map(|s| s.parse().map_err(|e: String| TickError::InvalidArgument(e))).transpose()?,
        priority: priority.map(|s| s.parse().map_err(|e: String| TickError::InvalidArgument(e))).transpose()?,
        parent_id,
        root_only: root,
        limit,
        offset,
    };
    db.list_issues(&filter)
}

pub fn show(db: &Database, id: i64) -> Result<IssueDetail> {
    let issue = db.get_issue(id)?;

    let parent = match issue.parent_id {
        Some(pid) => Some(db.get_issue_summary(pid)?),
        None => None,
    };
    let children = db.get_children(id)?;
    let (depends_on, depended_by) = db.list_links(id)?;
    let comments = db.list_comments(id)?;

    Ok(IssueDetail {
        issue,
        parent,
        children,
        depends_on,
        depended_by,
        comments,
    })
}

pub fn update(
    db: &Database, id: i64, title: Option<&str>, description: Option<&str>,
    issue_type: Option<&str>, priority: Option<&str>, parent_id: Option<i64>,
) -> Result<Issue> {
    let it = issue_type.map(|s| s.parse::<IssueType>().map_err(|e: String| TickError::InvalidArgument(e))).transpose()?;
    let p = priority.map(|s| s.parse::<Priority>().map_err(|e: String| TickError::InvalidArgument(e))).transpose()?;

    // Validate parent cycle if changing parent
    if let Some(pid) = parent_id {
        validators::validate_parent_no_cycle(db.conn(), id, pid)?;
    }

    let parent = parent_id.map(Some);
    db.update_issue_fields(id, title, description, it.as_ref(), p.as_ref(), parent)
}

pub fn start(db: &Database, id: i64, branch: &str) -> Result<Issue> {
    validators::validate_start(db.conn(), id, branch)?;
    db.update_issue_status_atomic(id, "open", "in-progress", None, Some(branch), false, false)
}

pub fn done(db: &Database, id: i64) -> Result<Issue> {
    db.update_issue_status_atomic(id, "in-progress", "done", None, None, false, false)
}

pub fn close(
    db: &Database, id: i64, comment: Option<&str>, role: &str,
    resolution: &str,
) -> Result<Issue> {
    let res: Resolution = resolution.parse().map_err(|e: String| TickError::InvalidArgument(e))?;
    let comment_role: CommentRole = role.parse().map_err(|e: String| TickError::InvalidArgument(e))?;

    // Get current issue to determine allowed resolution
    let issue = db.get_issue(id)?;

    // Validate resolution is allowed for current status
    validators::validate_close_resolution(&issue.status, &res)?;

    // Determine expected status for atomic update
    let expected = issue.status.to_string();
    let valid_from = matches!(issue.status, IssueStatus::Open | IssueStatus::InProgress | IssueStatus::Done);
    if !valid_from {
        return Err(TickError::Conflict(format!(
            "cannot close issue #{id} from status '{}'", issue.status
        )));
    }

    // Add comment if provided
    if let Some(body) = comment {
        db.create_comment(id, body, &comment_role)?;
    }

    db.update_issue_status_atomic(id, &expected, "closed", Some(resolution), None, false, false)
}

pub fn reopen(db: &Database, id: i64) -> Result<Issue> {
    db.update_issue_status_atomic(id, "closed", "open", None, None, true, true)
}

use clap::Parser;
use tick::cli::{Cli, Commands};
use tick::cli::issue::IssueCommands;
use tick::commands;
use tick::error::TickError;

fn main() {
    let cli = Cli::parse();

    if let Err(e) = run(cli) {
        let json = serde_json::json!({
            "error": e.to_string(),
            "code": e.error_code(),
        });
        eprintln!("{}", json);
        std::process::exit(e.exit_code());
    }
}

fn run(cli: Cli) -> tick::error::Result<()> {
    match cli.command {
        Commands::Init => {
            let db = commands::init::run(cli.db.as_deref())?;
            let version = db.schema_version()?;
            output(&cli, &serde_json::json!({"initialized": true, "schema_version": version}))?;
        }
        Commands::Version => {
            let version = serde_json::json!({
                "version": env!("CARGO_PKG_VERSION"),
                "schema_version": tick::db::migrate::expected_version(),
            });
            output(&cli, &version)?;
        }
        Commands::Status => {
            let db = commands::init::open_db(cli.db.as_deref())?;
            let counts = db.count_by_status()?;
            if cli.pretty {
                tick::output::pretty::print_status_counts(&counts);
            } else {
                tick::output::json::print(&counts);
            }
            return Ok(());
        }
        Commands::Issue(cmd) => {
            let db = commands::init::open_db(cli.db.as_deref())?;
            match cmd {
                IssueCommands::Create { title, description, issue_type, priority, parent } => {
                    let issue = commands::issue::create(&db, &title, &description, &issue_type, &priority, parent)?;
                    output_issue(&cli, &issue)?;
                }
                IssueCommands::List { status, issue_type, priority, parent, root, limit, offset } => {
                    let issues = commands::issue::list(
                        &db, status.as_deref(), issue_type.as_deref(),
                        priority.as_deref(), parent, root, limit, offset,
                    )?;
                    if cli.pretty {
                        tick::output::pretty::print_issue_list(&issues);
                    } else {
                        tick::output::json::print(&issues);
                    }
                    return Ok(());
                }
                IssueCommands::Show { id } => {
                    let detail = commands::issue::show(&db, id)?;
                    if cli.pretty {
                        tick::output::pretty::print_issue_detail(&detail);
                    } else {
                        tick::output::json::print(&detail);
                    }
                    return Ok(());
                }
                IssueCommands::Update { id, title, description, issue_type, priority, parent } => {
                    let issue = commands::issue::update(
                        &db, id, title.as_deref(), description.as_deref(),
                        issue_type.as_deref(), priority.as_deref(), parent,
                    )?;
                    output_issue(&cli, &issue)?;
                }
                IssueCommands::Start { id, branch } => {
                    let issue = commands::issue::start(&db, id, &branch)?;
                    output_issue(&cli, &issue)?;
                }
                IssueCommands::Done { id } => {
                    let issue = commands::issue::done(&db, id)?;
                    output_issue(&cli, &issue)?;
                }
                IssueCommands::Close { id, comment, role, resolution } => {
                    let issue = commands::issue::close(&db, id, comment.as_deref(), &role, &resolution)?;
                    output_issue(&cli, &issue)?;
                }
                IssueCommands::Reopen { id } => {
                    let issue = commands::issue::reopen(&db, id)?;
                    output_issue(&cli, &issue)?;
                }
            }
        }
    }
    Ok(())
}

fn output(cli: &Cli, value: &serde_json::Value) -> tick::error::Result<()> {
    if cli.pretty {
        println!("{}", serde_json::to_string_pretty(value).unwrap());
    } else {
        tick::output::json::print(value);
    }
    Ok(())
}

fn output_issue(cli: &Cli, issue: &tick::models::Issue) -> tick::error::Result<()> {
    if cli.pretty {
        tick::output::pretty::print_issue(issue);
    } else {
        tick::output::json::print(issue);
    }
    Ok(())
}
