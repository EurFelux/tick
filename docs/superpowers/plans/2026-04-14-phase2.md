pub fn list_by_issue(conn: &Connection, issue_id: i64, role: Option<&CommentRole>) -> Result<Vec<Comment>> {
    let (sql, params): (String, Vec<Box<dyn rusqlite::types::ToSql>>) = if let Some(r) = role {
        (
            "SELECT id, issue_id, body, role, created_at FROM comments WHERE issue_id = ?1 AND role = ?2 ORDER BY id".to_string(),
            vec![Box::new(issue_id), Box::new(r.to_string())],
        )
    } else {
        (
            "SELECT id, issue_id, body, role, created_at FROM comments WHERE issue_id = ?1 ORDER BY id".to_string(),
            vec![Box::new(issue_id)],
        )
    };
    let params_refs: Vec<&dyn rusqlite::types::ToSql> = params.iter().map(|b| b.as_ref()).collect();
    let mut stmt = conn.prepare(&sql)?;
    let comments = stmt
        .query_map(params_refs.as_slice(), |row| {
            Ok(Comment {
                id: row.get(0)?,
                issue_id: row.get(1)?,
                body: row.get(2)?,
                role: row.get::<_, String>(3)?.parse().expect("invalid role in DB"),
                created_at: row.get(4)?,
            })
        })?
        .collect::<std::result::Result<Vec<_>, _>>()?;
    Ok(comments)
}

pub fn create(conn: &Connection, from_id: i64, to_id: i64) -> Result<()> {
    conn.execute(
        "INSERT INTO issue_links (from_issue_id, to_issue_id, relation) VALUES (?1, ?2, 'depends-on')",
        params![from_id, to_id],
    )?;
    conn.execute(
        "INSERT INTO issue_links (from_issue_id, to_issue_id, relation) VALUES (?1, ?2, 'depended-by')",
        params![to_id, from_id],
    )?;
    Ok(())
}

pub fn delete(conn: &Connection, from_id: i64, to_id: i64) -> Result<()> {
    let deleted1 = conn.execute(
        "DELETE FROM issue_links WHERE from_issue_id = ?1 AND to_issue_id = ?2 AND relation = 'depends-on'",
        params![from_id, to_id],
    )?;
    conn.execute(
        "DELETE FROM issue_links WHERE from_issue_id = ?1 AND to_issue_id = ?2 AND relation = 'depended-by'",
        params![to_id, from_id],
    )?;
    if deleted1 == 0 {
        return Err(TickError::NotFound(format!("link from #{from_id} to #{to_id} not found")));
    }
    Ok(())
}

/// Get all issue IDs that depend on the given issue (for cascade wontfix)
pub fn get_depended_by_ids(conn: &Connection, issue_id: i64) -> Result<Vec<i64>> {
    let mut stmt = conn.prepare(
        "SELECT from_issue_id FROM issue_links WHERE to_issue_id = ?1 AND relation = 'depends-on'"
    )?;
    let ids = stmt.query_map([issue_id], |row| row.get(0))?
        .collect::<std::result::Result<Vec<i64>, _>>()?;
    Ok(ids)
}
```

/// Validate that a link can be created:
/// - No self-reference
/// - No cycle in dependency chain
/// - If from issue is non-open, to issue must be closed(resolved)
pub fn validate_link(db: &Database, from_id: i64, to_id: i64) -> Result<()> {
    if from_id == to_id {
        return Err(TickError::InvalidArgument("issue cannot depend on itself".to_string()));
    }

    // Check dependency cycle: walk from to_id's depends-on chain, if we reach from_id it's a cycle
    check_dependency_cycle(db, from_id, to_id)?;

    // If from issue is already past open, target must be resolved
    let from_issue = db.get_issue(from_id)?;
    if from_issue.status != IssueStatus::Open {
        let to_issue = db.get_issue(to_id)?;
        if to_issue.status != IssueStatus::Closed || to_issue.resolution != Some(Resolution::Resolved) {
            return Err(TickError::InvalidArgument(format!(
                "issue #{from_id} is '{}': dependency #{to_id} must be closed(resolved)",
                from_issue.status
            )));
        }
    }

    Ok(())
}

fn check_dependency_cycle(db: &Database, from_id: i64, to_id: i64) -> Result<()> {
    // BFS/DFS from to_id following depends-on links; if we reach from_id, it's a cycle
    let mut visited = std::collections::HashSet::new();
    let mut queue = vec![to_id];
    while let Some(current) = queue.pop() {
        if current == from_id {
            return Err(TickError::InvalidArgument(format!(
                "adding dependency #{from_id} → #{to_id} would create a cycle"
            )));
        }
        if visited.insert(current) {
            let (deps, _) = db.list_links(current)?;
            for dep in deps {
                queue.push(dep.id);
            }
        }
    }
    Ok(())
}

/// Cascade wontfix: recursively close all issues that depend on the given issue
pub fn cascade_wontfix(db: &Database, wontfixed_issue_id: i64) -> Result<()> {
    let dependent_ids = db.get_depended_by_ids(wontfixed_issue_id)?;
    for dep_id in dependent_ids {
        let dep = db.get_issue(dep_id)?;
        if dep.status == IssueStatus::Closed {
            continue; // already closed
        }
        // Force close regardless of current status
        db.update_issue_status_atomic(
            dep_id,
            &dep.status,
            &IssueStatus::Closed,
            Some(Some(&Resolution::Wontfix)),
            None,
            false,
            false,
        )?;
        db.create_comment(
            dep_id,
            &format!("Closed by cascade: dependency #{} was abandoned", wontfixed_issue_id),
            &crate::models::CommentRole::System,
        )?;
        // Recurse
        cascade_wontfix(db, dep_id)?;
    }
    Ok(())
}
```

pub fn link(db: &Database, from_id: i64, relation: &str, to_id: i64) -> Result<serde_json::Value> {
    if relation != "depends-on" {
        return Err(TickError::InvalidArgument(format!(
            "unknown relation '{}', only 'depends-on' is supported", relation
        )));
    }
    // Verify both issues exist
    db.get_issue(from_id)?;
    db.get_issue(to_id)?;
    validators::validate_link(db, from_id, to_id)?;
    db.create_link(from_id, to_id)?;
    Ok(serde_json::json!({"linked": true, "from": from_id, "to": to_id, "relation": "depends-on"}))
}

pub fn unlink(db: &Database, from_id: i64, to_id: i64) -> Result<serde_json::Value> {

    Ok(serde_json::json!({"unlinked": true, "from": from_id, "to": to_id}))
}

pub fn print_filtered<T: Serialize>(value: &T, fields: &[&str]) {
    let v = serde_json::to_value(value).unwrap();
    match v {
        serde_json::Value::Object(map) => {
            let filtered: serde_json::Map<String, serde_json::Value> = map
                .into_iter()
                .filter(|(k, _)| fields.contains(&k.as_str()))
                .collect();
            println!("{}", serde_json::to_string(&filtered).unwrap());
        }
        serde_json::Value::Array(arr) => {
            let filtered: Vec<serde_json::Value> = arr
                .into_iter()
                .map(|item| {
                    if let serde_json::Value::Object(map) = item {
                        let m: serde_json::Map<String, serde_json::Value> = map
                            .into_iter()
                            .filter(|(k, _)| fields.contains(&k.as_str()))
                            .collect();
                        serde_json::Value::Object(m)
                    } else {
                        item
                    }
                })
                .collect();
            println!("{}", serde_json::to_string(&filtered).unwrap());
        }
        _ => println!("{}", serde_json::to_string(&v).unwrap()),
    }
}

use std::io::BufRead;

pub fn batch_create(db: &Database) -> Result<Vec<serde_json::Value>> {
    let stdin = std::io::stdin();
    let mut results = Vec::new();
    let mut has_error = false;

    for (i, line) in stdin.lock().lines().enumerate() {
        let line = line.map_err(|e| TickError::Internal(anyhow::anyhow!("stdin read error: {}", e)))?;
        if line.trim().is_empty() {
            continue;
        }
        let parsed: std::result::Result<serde_json::Value, _> = serde_json::from_str(&line);
        match parsed {
            Ok(obj) => {
                let title = obj.get("title").and_then(|v| v.as_str()).unwrap_or("");
                if title.is_empty() {
                    results.push(serde_json::json!({"error": "title is required", "code": "INVALID_ARGUMENT", "line": i + 1}));
                    has_error = true;
                    continue;
                }
                let desc = obj.get("description").and_then(|v| v.as_str()).unwrap_or("");
                let itype = obj.get("type").and_then(|v| v.as_str()).unwrap_or("feature");
                let prio = obj.get("priority").and_then(|v| v.as_str()).unwrap_or("medium");
                let parent = obj.get("parent").and_then(|v| v.as_i64());

                match create(db, title, Some(desc), itype, prio, parent) {
                    Ok(issue) => results.push(serde_json::to_value(&issue).unwrap()),
                    Err(e) => {
                        results.push(serde_json::json!({"error": e.to_string(), "code": e.error_code(), "line": i + 1}));
                        has_error = true;
                    }
                }
            }
            Err(e) => {
                results.push(serde_json::json!({"error": format!("invalid JSON: {e}"), "code": "INVALID_ARGUMENT", "line": i + 1}));
                has_error = true;
            }
        }
    }

    if has_error {
        // Return results but signal partial failure via the caller
    }
    Ok(results)
}

use crate::error::{Result, TickError};
use rusqlite::{params, Connection};
use serde::Serialize;

#[derive(Debug, Serialize)]
pub struct ConfigEntry {
    pub key: String,
    pub value: String,
}

pub fn set(conn: &Connection, key: &str, value: &str) -> Result<()> {
    conn.execute(
        "INSERT OR REPLACE INTO config (key, value) VALUES (?1, ?2)",
        params![key, value],
    )?;
    Ok(())
}

pub fn get(conn: &Connection, key: &str) -> Result<ConfigEntry> {
    conn.query_row(
        "SELECT key, value FROM config WHERE key = ?1",
        [key],
        |row| Ok(ConfigEntry { key: row.get(0)?, value: row.get(1)? }),
    )
    .map_err(|e| match e {
        rusqlite::Error::QueryReturnedNoRows => {
            TickError::NotFound(format!("config key '{}' not found", key))
        }
        other => TickError::Db(other),
    })
}

pub fn list(conn: &Connection) -> Result<Vec<ConfigEntry>> {
    let mut stmt = conn.prepare("SELECT key, value FROM config ORDER BY key")?;
    let entries = stmt
        .query_map([], |row| Ok(ConfigEntry { key: row.get(0)?, value: row.get(1)? }))?
        .collect::<std::result::Result<Vec<_>, _>>()?;
    Ok(entries)
}
